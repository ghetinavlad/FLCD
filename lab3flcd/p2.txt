func main() {
  int n;
  read(n);
  bool ok;
  ok = true;
  if (n < 2 or n > 2 and n % 2 == 0) {
    ok=0;
  }
  if (ok == 1) {
    int d;
    for (d = 3; d * d <= n; d = d + 2){
      if (n % d == 0) {
        ok=0;
      }
    }
  }
  if (ok == 1) {
    print('is prime');
  }
  else {
    print('is not prime');
  }
}
